001package de.yafp.gimmepassword;
002
003import android.os.Bundle;
004import android.support.annotation.NonNull;
005import android.support.annotation.Nullable;
006import android.support.v4.app.Fragment;
007import android.text.InputFilter;
008import android.util.Log;
009import android.view.LayoutInflater;
010import android.view.View;
011import android.view.ViewGroup;
012import android.widget.EditText;
013
014/**
015 * Password Generation Tab - Typ; Default
016 */
017public class TabDefault extends Fragment{
018
019    private static final String TAG = "Gimme Password";
020
021    @Nullable
022    @Override
023    public View onCreateView(@NonNull LayoutInflater inflater, ViewGroup container, Bundle savedInstanceState) {
024
025        // Init some UI Items
026        //
027        Log.d(TAG,"...preparing tab1: default");
028
029        // get view
030        View view = inflater.inflate(R.layout.tab_default, container, false);
031
032        // define min- and max- value for password length
033        EditText n_passwordLength = view.findViewById(R.id.t1_passwordLength);
034        n_passwordLength.setFilters(new InputFilter[]{new InputFilterMinMax("1", "64")});
035
036        return view;
037    }
038}
