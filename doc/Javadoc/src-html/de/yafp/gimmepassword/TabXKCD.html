001package de.yafp.gimmepassword;
002
003import android.os.Bundle;
004import android.support.annotation.NonNull;
005import android.support.annotation.Nullable;
006import android.support.v4.app.Fragment;
007import android.text.InputFilter;
008import android.util.Log;
009import android.view.LayoutInflater;
010import android.view.View;
011import android.view.ViewGroup;
012import android.widget.EditText;
013import android.widget.Spinner;
014
015import java.util.Locale;
016
017
018/**
019 * Password Generation Tab - Typ; XKCD
020 */
021public class TabXKCD extends Fragment{
022
023    private static final String TAG = "Gimme Password";
024
025    @Nullable
026    @Override
027    public View onCreateView(@NonNull LayoutInflater inflater, ViewGroup container, Bundle savedInstanceState) {
028
029        // Init some UI Items
030        //
031        Log.d(TAG,"...preparing tab2: XKCD");
032
033        // get view
034        View view = inflater.inflate(R.layout.tab_xkcd, container, false);
035
036        // define min- and max- value for password length
037        EditText n_passwordLength = view.findViewById(R.id.t2_passwordLength);
038        n_passwordLength.setFilters(new InputFilter[]{new InputFilterMinMax("1", "10")});
039
040        // detect device language
041        String deviceLanguage;
042        deviceLanguage =  Locale.getDefault().getISO3Language(); // get 3 letter ISO code
043        Log.v(TAG,"Device language is: "+deviceLanguage+" (ISO-3).");
044
045        // pre-select a language in language-spinner if possible
046        Spinner t2_s_language_selection = view.findViewById(R.id.t2_languageSelection);
047        switch (deviceLanguage) {
048            case "fin":
049                Log.v(TAG, "...Pre-selecting finnish");
050                t2_s_language_selection.setSelection(1);
051                break;
052            case "deu":
053                Log.v(TAG, "...Pre-selecting german");
054                t2_s_language_selection.setSelection(2);
055                break;
056            case "ita":
057                Log.v(TAG, "...Pre-selecting italian");
058                t2_s_language_selection.setSelection(3);
059                break;
060            case "jap":
061                Log.v(TAG, "...Pre-selecting japanese");
062                t2_s_language_selection.setSelection(4);
063                break;
064            case "esp":
065                Log.v(TAG, "...Pre-selecting spanish");
066                t2_s_language_selection.setSelection(5);
067                break;
068            default:
069                Log.v(TAG, "...Pre-selecting english (default)");
070                t2_s_language_selection.setSelection(0);
071                break;
072        }
073        return view;
074    }
075}
