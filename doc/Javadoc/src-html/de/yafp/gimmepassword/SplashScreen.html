001package de.yafp.gimmepassword;
002
003import android.app.Activity;
004import android.content.Intent;
005import android.os.Bundle;
006import android.util.Log;
007
008/**
009 * Used to handle the pseudo splashscreen on app launch
010 */
011public class SplashScreen extends Activity {
012
013    private static final String TAG = "Gimme Password";
014
015    private final int _splashTime = 500; // time to display the splash screen in ms - default 1000
016
017    @Override
018    public void onCreate(Bundle savedInstanceState) {
019        super.onCreate(savedInstanceState);
020        setContentView(R.layout.splash);
021
022        Thread splashTread = new Thread() {
023            @Override
024            public void run() {
025                try {
026                    int waited = 0;
027                    while (waited < _splashTime) {
028                        sleep(100);
029                        waited += 100;
030                    }
031                } catch (Exception e) {
032                    Log.e(TAG, "Catching exception in SlashScreen");
033
034                } finally {
035
036                    startActivity(new Intent(SplashScreen.this,
037                            de.yafp.gimmepassword.GimmePassword.class));
038                    finish();
039                }
040            }
041        };
042        splashTread.start();
043    }
044}
